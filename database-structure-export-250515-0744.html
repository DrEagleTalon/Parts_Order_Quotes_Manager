View: active_purchase_orders 
SELECT po.purchase_order_number,
                           po.date_purchased,
                           po.status,
                           c.company_name,
                           c.contact_email
                      FROM purchase_orders po
                           JOIN
                           companies c ON po.company = c.company_name
                     WHERE po.status IN ('Pending', 'Approved')


View: inventory_summary 
SELECT p.model_number,
                           p.name,
                           p.qty AS stock_quantity,
                           p.price_per_unit,
                           (p.qty * p.price_per_unit) AS total_value
                      FROM parts p
                     WHERE p.qty > 0


View: linked_quotes_po 
SELECT po.purchase_order_number,
                           q.quote_number,
                           q.company,
                           q.estimated_total_cost,
                           po.total_cost
                      FROM purchase_orders po
                           JOIN
                           quotes q ON po.quote_number = q.quote_number
